    <div id="output"></div>

    <script>
        async function kirimKeAI() {
            const input = document.getElementById('userInput').value;
            const outputBox = document.getElementById('output');
            
            if (!input) return alert('Ketik sesuatu dulu, bos!');
            
            outputBox.style.display = 'block';
            outputBox.innerText = 'Sedang berpikir...';

            try {
                // Kita menembak ke '/api/generate', bukan langsung ke Google. Ini rahasia keamanannya!
                const response = await fetch('/api/generate', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ prompt: input })
                });

                const data = await response.json();
                outputBox.innerText = data.jawaban || "Waduh, ada masalah.";
            } catch (error) {
                outputBox.innerText = 'Error: Gagal terhubung ke server.';
            }
        }
    </script>
